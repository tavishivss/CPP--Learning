Various types of programming, Area and perimeter of a rectangle;

1)Monolithic programming:-

#include<iostream>
using namespace std;

int main()
{
    int a,p;
    int l,b;
    cin>>l>>b;
    a=l*b;
    p=2*(l+b);
    cout<<p<<a;
}

2)Modular programming:-

//monolithic program 

#include<iostream>
using namespace std;

int area(int l, int b)
{
    return l*b;
}
int peri(int l, int b)
{
    return 2*(l + b);
}
int main()
{
    int l,b;
    cin>>l>>b;
    area(l,b);
    peri(l,b);
}
