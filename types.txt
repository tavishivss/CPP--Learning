Various types of programming, Area and perimeter of a rectangle;

1)Monolithic programming:-

#include<iostream>
using namespace std;

int main()
{
    int a,p;
    int l,b;
    cin>>l>>b;
    a=l*b;
    p=2*(l+b);
    cout<<p<<a;
}

2)Modular programming:-

//monolithic program 

#include<iostream>
using namespace std;

int area(int l, int b)
{
    return l*b;
}
int peri(int l, int b)
{
    return 2*(l + b);
}
int main()
{
    int l,b;
    cin>>l>>b;
    area(l,b);
    peri(l,b);
}

3)Structures using functions

#include<iostream>
using namespace std;

struct rectangle
{ 
   int a,b; 
};

int area(struct rectangle r)
{
   int area;
   area = r.a*r.b;
   cout<<area<<endl;
   
}

int peri(struct rectangle r)
{
   int peri;
   peri=2*(r.a+r.b);
   cout<<peri;
}

int main()
{
    struct rectangle r;
    r.a=10;
    r.b=20;
    area(r);
    peri(r); 
}

#include<iostream>
#include<stdio.h>
using namespace std;

struct rectangle
{ 
   int a,b; 
};

void init(struct rectangle *r,int l,int b)
{
    r->a=l;
    r->b=b;
}

int area(struct rectangle r)
{
    return r.a*r.b;
   
}

int peri(struct rectangle r)
{
    return 2*(r.a+r.b);
}

int main()
{
    int l,b;
    rectangle r={0,0};
    cin>>l>>b;
    init(&r,l,b);
    int a=area(r);
    int p=peri(r);
    cout<<a<<p;
    return 0;   
}
